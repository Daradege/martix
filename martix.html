131    async def start(self) -> None:
132        """
133        Start the Matrix client and begin processing events.
134        
135        Raises:
136            AuthenticationError: If login fails
137            NetworkError: If connection fails
138        """
139        if self.e2ee:
140            store = SqliteStore(
141                user_id=self.user_id,
142                device_id=self.device_name,
143                store_path=str(self.store_path)
144            )
145            self._client = AsyncClient(
146                homeserver=self.homeserver,
147                user=self.user_id,
148                device_id=self.device_name,
149                store_path=str(self.store_path)
150            )
151        else:
152            self._client = AsyncClient(
153                homeserver=self.homeserver,
154                user=self.user_id,
155                device_id=self.device_name
156            )
157        
158        try:
159            response = await self._client.login(self.password, device_name=self.device_name)
160            if not isinstance(response, LoginResponse):
161                raise AuthenticationError(f"Login failed: {response}")
162                
163            self._user = create_user_object(self.user_id, self._client)
164            
165            if self.e2ee:
166                await self._setup_e2ee()
167            
168            # Trust all devices by default to avoid verification issues
169            if self.e2ee:
170                self._client.trust_devices = True
171            
172            self._load_sync_token()
173            
174            self._client.add_event_callback(self._handle_message, (RoomMessageText, MegolmEvent))
175            self._client.add_event_callback(self._handle_invite, InviteEvent)
176            self._client.add_event_callback(self._handle_member_event, RoomMemberEvent)
177            
178            await self._event_handler.trigger_ready()
179            
180            await self._sync_forever()
181            
182        except Exception as e:
183            if isinstance(e, (AuthenticationError, NetworkError)):
184                raise
185            raise NetworkError(f"Failed to start client: {e}")
186        finally:
187            if self._client:
188                await self._client.close()

211        """
212        Send a text message to a room.
213
214        Args:
215            room_id: Target room ID
216            text: Message text
217            reply_to: Event ID to reply to (optional)
218            parse_mode: How to parse message text, default is types.ParseTypes.MARKDOWN (optional)
219
220        Returns:
221            Event ID of the sent message
222        """
223
224        content = {
225            "msgtype": "m.text",
226            "body": text.strip()
227        }
228
229        if parse_mode != ParseTypes.TEXT:
230            content["format"] = "org.matrix.custom.html"
231
232            if parse_mode == ParseTypes.MARKDOWN:
233                content["formatted_body"] = markdown_to_html(text.strip())
234            elif parse_mode == ParseTypes.HTML:
235                content["formatted_body"] = text
236
237        if reply_to:
238            content["m.relates_to"] = {"m.in_reply_to": {"event_id": reply_to}}
239
240        response = await self.client.room_send(room_id, "m.room.message", content)
241        return response.event_id

330    def _load_sync_token(self) -> None:
331        """Load the sync token from file to resume from last position."""
332        if self._sync_token_file.exists():
333            try:
334                with open(self._sync_token_file, 'r') as f:
335                    data = json.load(f)
336                    self._next_batch = data.get('next_batch')
337            except (json.JSONDecodeError, KeyError):
338                pass

340    def _save_sync_token(self, token: str) -> None:
341        """Save the sync token to file."""
342        with open(self._sync_token_file, 'w') as f:
343            json.dump({'next_batch': token}, f)

345    async def _sync_forever(self) -> None:
346        """Continuously sync with the homeserver."""
347        while True:
348            try:
349                response = await self.client.sync(since=self._next_batch, timeout=30000)
350                if isinstance(response, SyncResponse):
351                    self._next_batch = response.next_batch
352                    self._save_sync_token(self._next_batch)
353                    
354                    if self.e2ee:
355                        await self._handle_key_verification(response)
356                        
357            except Exception as e:
358                print(f"Sync error: {e}")
359                await asyncio.sleep(5)
360                
361    async def _handle_key_verification(self, response: SyncResponse) -> None:
362        """Handle key verification after sync."""
363        if not self.e2ee or not self._client:
364            return
365            
366        # Upload keys if needed
367        if self._client.should_upload_keys:
368            await self._client.keys_upload()
369        
370        # Query keys for users in rooms to get device info
371        try:
372            users_to_query = set()
373            for room in self._client.rooms.values():
374                users_to_query.update(room.users)
375            
376            # if users_to_query:
377            #     await self._client.keys_query()
378                
379            # Auto-verify devices after querying
380            for user_id in self._client.device_store.users:
381                user_devices = self._client.device_store[user_id]
382                for device in user_devices.values():
383                    if not device.verified and not device.blacklisted:
384                        try:
385                            self._client.verify_device(device)
386                        except Exception as e:
387                            print(f"Could not verify device {device.id} for {user_id}: {e}")
388        except Exception as e:
389            print(f"Key verification error: {e}")

391    async def _handle_message(self, room: MatrixRoom, event) -> None:
392        """Handle incoming message events."""
393        if event.sender == self.user_id:
394            return
395            
396        if isinstance(event, MegolmEvent):
397            try:
398                decrypted_event = await self._client.decrypt_event(event)
399                if hasattr(decrypted_event, 'body'):
400                    event = decrypted_event
401                else:
402                    print(f"Could not decrypt event from {event.sender}")
403                    return
404            except Exception as e:
405                print(f"Decryption failed for event from {event.sender}: {e}")
406                # Continue processing even if decryption fails
407                if not hasattr(event, 'body'):
408                    return
409                
410        if not hasattr(event, 'body') or not event.body:
411            return
412            
413        message = create_message_object(room, event, self.client)
414        
415        if message.text.startswith(self.command_prefix):
416            command = parse_command(message, self.command_prefix)
417            if command:
418                await self._event_handler.trigger_command(command.name, command)
419        else:
420            await self._event_handler.trigger_message(message)
421            
422    async def _handle_invite(self, room: MatrixRoom, event: InviteEvent) -> None:
423        """Handle room invite events."""
424        await self._event_handler.trigger_invite(room, event)
425        
426    async def _handle_member_event(self, room: MatrixRoom, event: RoomMemberEvent) -> None:
427        """Handle room member events."""
428        if event.membership == "join":
429            await self._event_handler.trigger_member_join(room, event)
430        elif event.membership == "leave":
431            await self._event_handler.trigger_member_leave(room, event)

 18    """
 19    
 20    def __init__(self):
 21        """Initialize the event handler system."""
 22        self._ready_handlers: List[Callable] = []
 23        self._message_handlers: List[Callable] = []
 24        self._command_handlers: Dict[str, List[Callable]] = {}
 25        self._invite_handlers: List[Callable] = []
 26        self._member_join_handlers: List[Callable] = []
 27        self._member_leave_handlers: List[Callable] = []
 28        
 29    def on_ready(self, func: Callable) -> Callable:
 30        """
 31        Register a ready event handler.
 32        
 33        Args:
 34            func: Handler function
 35            
 36        Returns:
 37            The original function
 38        """
 39        self._ready_handlers.append(func)
 40        return func
 41        
 42    def on_message(self, func: Callable) -> Callable:
 43        """
 44        Register a message event handler.
 45        
 46        Args:
 47            func: Handler function that takes a Message parameter
 48            
 49        Returns:
 50            The original function
 51        """
 52        self._message_handlers.append(func)
 53        return func
 54        
 55    def on_command(self, command_name: str) -> Callable:
 56        """
 57        Register a command event handler.
 58        
 59        Args:
 60            command_name: Name of the command to handle
 61            
 62        Returns:
 63            Decorator function
 64        """
 65        def decorator(func: Callable) -> Callable:
 66            if command_name not in self._command_handlers:
 67                self._command_handlers[command_name] = []
 68            self._command_handlers[command_name].append(func)
 69            return func
 70        return decorator

 73        """

 93            The original function
